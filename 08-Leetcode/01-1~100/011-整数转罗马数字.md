# 整数转罗马数字

## 一、题目

[整数转罗马数字](https://leetcode-cn.com/problems/integer-to-roman/)

## 二、题解

```go
var valueSymbols = []struct {
	value  int
	symbol string
}{s
	{1000, "M"},
	{900, "CM"},
	{500, "D"},
	{400, "CD"},
	{100, "C"},
	{90, "XC"},
	{50, "L"},
	{40, "XL"},
	{10, "X"},
	{9, "IX"},
	{5, "V"},
	{4, "IV"},
	{1, "I"},
}

func intToRoman(num int) string {
	romanNum := []byte{}
	for _, vs := range valueSymbols{
		for num >= vs.value {
			num -= vs.value
			romanNum = append(romanNum, vs.symbol...)
		}
		if num == 0 {
			break
		}
	}
	return string(romanNum)
}
```
